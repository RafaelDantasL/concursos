document.getElementById('salvar').addEventListener('click', function() { const courseId = new URLSearchParams(window.location.search).get('id') || Date.now().toString(); const courseData = { nome: document.getElementById('nome').value, imagemUrl: document.getElementById('imagem-url').value, descricao: document.getElementById('descricao').value, preco: document.getElementById('preco').value, promocao: document.getElementById('promocao').value, parcelamento: document.getElementById('parcelamento').value, valorParcelamento: document.getElementById('valor-parcelamento').value, show: true }; fetch(`${firebaseConfig.databaseURL}/cursos/${courseId}.json`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(courseData) }).then(response => response.json()) .then(data => { alert('Curso salvo com sucesso!'); // Opcional: Redirecionar ou atualizar a p√°gina }); }); document.getElementById('criar-pagina-vendas').addEventListener('click', function() { window.location.href = `purchase_page.html?id=${courseId}`; });
